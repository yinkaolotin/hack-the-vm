#define _LARGEFILE64_SOURCE
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <fcntl.h>
#include <sys/types.h>

int main(int argc, char *argv[]) {
    if (argc != 3) {
        fprintf(stderr, "Usage: %s <pid> <address>\n", argv[0]);
        exit(1);
    }

    pid_t pid = atoi(argv[1]);
    unsigned long long addr = strtoull(argv[2], NULL, 16);

    char filename[64];
    snprintf(filename, sizeof(filename), "/proc/%d/mem", pid);

    int fd = open(filename, O_RDONLY);
    if (fd == -1) {
        perror("open");
        exit(1);
    }

    if (lseek64(fd, (off64_t)addr, SEEK_SET) == -1) {
        perror("lseek64");
        close(fd);
        exit(1);
    }

    char buffer[16];
    ssize_t bytes_read = read(fd, buffer, sizeof(buffer));
    if (bytes_read == -1) {
        perror("read");
        close(fd);
        exit(1);
    }

    printf("Read %zd bytes from address 0x%llx:\n", bytes_read, addr);
    for (ssize_t i = 0; i < bytes_read; i++) {
        printf("%02x ", (unsigned char)buffer[i]);
    }
    printf("\n");

    close(fd);
    return 0;
}
